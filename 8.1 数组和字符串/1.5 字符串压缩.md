
## 题目
利用字符重复出现的次数，编写一个方法，实现基本的字符串压缩功能。比如，字符串“aabcccccaa”会变为“a2b1c5a3”。若“压缩”后的字符串没有变短，则返回原先的字符串。

## 注解
直接模拟即可。


```python
# coding=utf-8


class Solution:

    def compress_str(self, temp_str):

        if temp_str is None or len(temp_str) < 2:
            return temp_str

        container = {}
        result = ''
        is_same = True

        for item in temp_str:

            if item in container:
                container[item] += 1
            else:
                for key, value in container.items():
                    result += key + str(value)
                    if value > 1:
                        is_same = False

                container = {}
                container[item] = 1

        for key, value in container.items():
            result += key + str(value)
            if value > 1:
                is_same = False

        if is_same is True:
            return temp_str
        return result

data_provider = [
    ['', ''],
    ['abcdefg', 'abcdefg'],
    ['aabcccccaaa', 'a2b1c5a3'],
    ['aaaaaaaaaaaaaa', 'a14'],
    ['abcdefghbbbbb', 'abcdefghbbbbb'],
    ['a', 'a'],
    ['ab', 'ab']
]


solution = Solution()

for value in data_provider:

    if solution.compress_str(value[0]) == value[1]:
        print('用例通过')
    else:
        print('用例不通过')

```
